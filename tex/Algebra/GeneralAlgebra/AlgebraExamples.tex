\begin{subsection}{Examples of Groups}
\begin{subsubsection}{$\ZZ/m\ZZ$}
    The additive group
    \mth{\ZZ/m\ZZ=\{\orl{0},\orl{1},\cdots,\orl{m-1}\}}
    \mth{\orl{a}+\orl{b}=\orl{a+b}}
    has the following properties
    \enu{
        \item $\ZZ/m\ZZ$ is cyclic.
        \item $\orl{a},\,(a,m)=1$ is a generator. In particular, $\ZZ/m\ZZ=\bkt{\orl{1}}$.
        \item Any subgroup of $\ZZ/m\ZZ$ is of the form $\bkt{\orl{d}}=\orl{d}\orl{\ZZ}\,(d\divd n).$
        \item $\Aut(\ZZ/m\ZZ)\cong(\ZZ/m\ZZ)^*$
        \item It can be decomposed using CRT.
    }
\end{subsubsection}
\begin{subsubsection}{$(\ZZ/m\ZZ)^*$}
    The multiplicative group
    \mth{(\ZZ/m\ZZ)^*=\{\orl{a}:(a,m)=1\}}
    \mth{\orl{a}\orl{b}=\orl{ab}}
    has the following properties
    \enu{
        \item $(\ZZ/p\ZZ)^*$ has order $\vphi(m)$.
        \item If $m=p$ is a prime number, then $(\ZZ/p\ZZ)^*$ is a cyclic group of order $p-1$.
    }
\end{subsubsection}
\begin{subsubsection}{Klein four-group}
    The group
    \mth{V=\{a,b:a^2=b^2=(ab)^2=1\}}
    has the following properties
    \enu{
        \item Each element is the inverse of itself.
        \item The product of two different nonidentity elements is the third.
    }
\end{subsubsection}
\begin{subsubsection}{Symmetric group $S_n$}
The group $S_n$ of all permutations of $n$ objects has the following properties
\enu{
    \item $S_n$ has order $n$.
    \item $S_n$ is not abelian when $n\geqs3$.
    \item $Z(S_n)=\{(1)\}$ for $n\geqs3$.
    \item The alternating group $A_n$ is a normal subgroup of $S_n$
    \item $S_n,\,A_n$ are unsolvable for $n\geqs5$.
    \item The coset decomposition of $S_n$ with respect to $S_{n-1}=\{\sigma\in S_n:\sigma(n)=n\}$ is
    \item Any $\sigma\in S_n$ is a product of some nonintersecting cyclic permutations.
}
The cases $n=3,4$ are as follows:
\desc{
    \item[$S_3$]The group $S_3=\{(1),(12),(13),(23),(123),(132)\}$ is a nonabelian group of order $6$. It is isomorphic to $D_6$ (See below for its presentation). The lattice of its subgroups is:
        \mth{\xymatrix{
        6& &	S_3&	&	\\
    	3& \uuline{\bkt{(123)}}\ar@{-}[ur]&	&	&	\\
	    2& &	\bkt{(12)}\ar@{-}[uu]&	\bkt{(13)}\ar@{-}[uul]&	\bkt{(12)}\ar@{-}[uull]\\
    	1& &	\{(1)\}\ar@{-}[uul]\ar@{-}[u]\ar@{-}[ur]\ar@{-}[urr]&	&	
        }}
    \item[$S_4$]The group $S_4$ is a nonabelian group of order $24$.
        \mtha{S_4&=&\{(1),(12),(13),(23),(123),(132),\\
        &=&(34),(12)(34),(143),(243),(1243),(4321),\\
        &=&(24),(13)(24),(234),(142),(1342),(1423),\\
        &=&(14),(14)(23),(134),(124),(1234),(1324)\}}
        It is isomorphic to the symmetry group of a regular tetrahedron.
    }
\end{subsubsection}
\begin{subsubsection}{Alternating group $A_n$}
The group $A_n$ of all even permutations of $n$ objects has the following properties
\enu{
    \item $A_n$ is generated by all $3$-cycles.
    \item For $n\geqs5$, all $3$-cycles are conjugate in $A_n$.
    \item For $n\geqs5$, $A_n$ is a simple group.
    \item For $n\geqs5$, $A_n$ is unsolvable.
}
\end{subsubsection}
\begin{subsubsection}{Dihedral group $D_{2n}$}
The group $D_{2n}$ of all symmetries of a $n$-sided regular polygon has the following properties
\enu{
    \item $D_{2n}$ is of order $2n$. Its elements are $n$ rotations and $n$ reflections.
    \item $D_{2n}$ has the group presentation
        \mth{D_{2n}=\{\rho,\vphi:\rho^n=1,\vphi^2=1,\rho\vphi=\vphi\rho^{n-1}\}}
    \item The center of $D_{2n}$ is
        \mth{Z(D_{2n})=\left\{\arr{{lll}\{1\}&,&n\ttt{ is odd}\\\{1,r=\rho^{n/2}\}&,&n\ttt{ is even}}\right.}
}
The case $n=4$ is as follows:
\desc{
    \item[$D_8$]The group $D_8$ of all symmetries of a square is a nonabelian group of order $8$. The lattice of its subgroups is:
        \mth{\xymatrix{
	    8&	&	&	D_8&	&	\\
	    4&	\uudl{\bkt{\rho}}\ar@{-}[urr]&	&	\udl{\bkt{\vphi,\rho^2}}\ar@{-}[u]&	&	\udl{\bkt{\vphi\rho,\rho^2}}\ar@{-}[ull]\\
	    2&	\wudl{\bkt{\rho^2}}\ar@{-}[u]\ar@{-}[urr]\ar@{-}[urrrr]&	\bkt{\vphi}\ar@{-}[ur]&\bkt{\vphi\rho}\ar@{-}[urr]&\bkt{\vphi\rho^2}\ar@{-}[ul]&\bkt{\vphi\rho^3}\ar@{-}[u]\\
    	1&	&	&	\{1\}\ar@{-}[ull]\ar@{-}[ul]\ar@{-}[u]\ar@{-}[ur]\ar@{-}[urr]&	&	
        }}
    }
\end{subsubsection}
\begin{subsubsection}{Quaternion group $Q_8$}
The quaternion group $Q_8=\{1,-1,i,-i,j,-j,k,-k\}$ is a nonabelian group of order $8$. It has the group presentation
\mth{Q_8=\bkt{-1,i,j,k:(-1)^2=1,i^2=j^2=k^2=ijk=-1}}
The lattice of its subgroups is:
\mth{\xymatrix{
	8&	&	Q_8&	\\
	4&	\udl{\{1,-1,i,-i\}}\ar@{-}[ur]&	\udl{\{1,-1,j,-j\}}\ar@{-}[u]&	\udl{\{1,-1,k,-k\}}\ar@{-}[ul]\\
	2&	&	\uudl{\wudl{\{1,-1\}}}\ar@{-}[u]\ar@{-}[ul]\ar@{-}[ur]&	\\
	1&	&	\{1\}\ar@{-}[u]&	
    }}
\end{subsubsection}
\begin{subsubsection}{General linear group (over a field/ring) $GL_n$}
The general linear group of degree $n$ over a field $F$ or a ring $R$ is the set of $n\x n$ invertible matrices with entries from $F$ or $R$, associated with matrix multiplication. It is denoted $GL_n(F)$ or $GL_n(R)$.
\end{subsubsection}
\begin{subsubsection}{Special linear group $SL_n$}
The special linear group $SL_n(F)$ of degree $n$ over a field $F$ is the subgroup of $GL_n(F)$ in which each matrix has determinant $1$.
\end{subsubsection}
\begin{subsubsection}{Orthogonal group $O_n$}
The orthogonal group $O_n$ is a subgroup of $GL_n$ in which each matrix is an orthogonal matrix.
\end{subsubsection}
\begin{subsubsection}{Special orthogonal group $SO_n$}
The special orthogonal group $SO_n$ is a subgroup of $O_n$ of index $2$ in which each matrix is an orthogonal matrix with determinant $1$. It is also called the rotation group.
\end{subsubsection}
\begin{subsubsection}{Unitary group $U_n$}
The unitary group $U_n$ of degree $n$ is a subgroup of $GL_n(\CC)$ in which each matrix is a unitary matrix.
\end{subsubsection}
\begin{subsubsection}{Special unitary group $SU_n$}
The special unitary group $SU_n$ of degree $n$ is a subgroup of $U_n$ in which each matrix is a unitary matrix with determinant $1$.
\end{subsubsection}
