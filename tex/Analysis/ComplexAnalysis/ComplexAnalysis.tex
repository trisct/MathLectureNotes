\begin{section}{Geometry of the Complex Plane}
\begin{subsection}{Definitions}
\begin{description}
\item[The Riemann sphere\index{sphere!Riemann}]There are numerous ways to define the Riemann sphere.
    \itm{
        \item[(i)]As a complex manifold, \tf{the Riemann sphere} is a separate compact topological space $\PP^1=\CC\cup\{\infty\}$ together with two charts $\vphi_1,\vphi_2$. In $\PP^1$ an open set is either an open set in $\CC$, or a set of the form $V\cup\{\infty\}$ where $V$ is the complement of a compact set. The charts $\vphi_1,\vphi_2$ are:
            \[\arr{{ccrcl}\vphi_1&:&U_1=\PP^1\backslash\{\infty\}=\CC&\to&\CC\\
                &&z&\mapsto&z\\
                \vphi_2&:&U_2=\PP^1\backslash\{0\}=\CC^*\cup\{\infty\}&\to&\CC\\
                &&z&\mapsto&1/z}\]
        \item[(ii)]As a sphere, \tf{the Riemann sphere} can be embedded in $\RR^3$ as $S^2$ using a projection:
            \mtha{\orl\CC&\to&S^2\\
                z&\mapsto&\prn{\frac{z+\orl z}{\abs{z}^2+1},\frac{z-\orl z}{i(\abs{z}^2+1)},\frac{\abs{z}^2-1}{\abs{z}^2+1}}\\
                \infty&\mapsto&(0,0,1)}
            The projection is invertible, and the inverse is
            \mtha{S^2&\to&\orl\CC\\
                (x,y,z)&\mapsto&\frac{x+iy}{1-z}\\
                (0,0,1)&\mapsto&\infty}
    }
\end{description}
\end{subsection}

\begin{subsection}{Propositions}
\begin{description}
\item[Symmetric points]\
    \itm{
        \item[(i)]Let $Az\orl z+\orl Bz+B\orl z+C=0$ be a circle or a line, then for $z_1,z_2\in\CC$,
            \[Az_1\orl z_2+\orl Bz_1+B\orl z_2+C=0\iff z_1,z_2\ttt{ are symmtric about the circle}\]
        \item[(ii)]Let $S^2$ be the Riemann sphere, then $z,\frac{1}{\orl z}$ correspond to points on the sphere symmetric about the $xOy$ plane
        \item[(iii)]Let $S^2$ be the Riemann sphere, then for $z_1,z_2\in\CC$,
            \[z_1\orl z_2=-1\iff z_1,z_2\ttt{ correspond to endpoints of a diameter of} S^2\]
    }
\end{description}
\end{subsection}
\end{section}

\begin{section}{Holomorphic Functions}
\begin{subsection}{Definitions}
\begin{description}
\item[Derivative\index{derivative}]Let $\Omega$ be a open set. The \tf{derivative} of a complex function $f$ at a point $z_0\in\Omega$ is a complex number $f'(z_0)$ that satisfies either one of the equivalent conditions.
    \itm{
        \item[(i)]$f'(z_0)=\lim\limits_{z\to z_0}\frac{f(z)-f(z_0)}{z-z_0}$;
        \item[(ii)]$f(z)=f(z_0)+f'(z_0)(z-z_0)+o(z-z_0)$.
    }
    If such $f'(z_0)$ exists, we say $f$ is differentiable at $z_0$. If the limit in (i) does not exist, or that there is no such number satisfying  (ii), we say that $f$ is \tf{not differentiable}\index{differentiable} at $z_0$.
\item[Holomorphic, analytic\index{function!holomorphic}\index{function!analytic}]Let $f$ be a complex function on an open set $\Omega$. For $z_0\in\Omega$, if $f$ is analytic in a neighborhood of $z_0$, we say $f$ is \tf{analytic at $z_0$}. If $f$ is analytic at each $z\in\Omega$, or equivalently, $f$ is differentiable everywhere in $\Omega$, we say $f$ is \tf{holomorphic} or \tf{analytic} in $\Omega$. The set of holomorphic function in $\Omega$ is denoted by $H(\Omega)$. A function holomorphic on $\CC$ is called an \tf{entire function} or \tf{integral function}\index{function!entire}\index{function!integral}.
\item[Power series\index{series!power}\index{power series}]A \tf{power series} is of the form
    \[s(z)=\sum_{k=0}^\infty a_k(z-z_0)^k\]
    To each power series there corresponds a number
    \[R=\frac{1}{\limsup_n\sqrt[n]{\abs{a_n}}}\in[0,\infty]\]
    called the \tf{radius of convergence}\index{radius!convergence} such that
    \itm{
        \item[(i)]In $B(z_0,R)$, $s$ converges. More specifically, $\forall r<R,\ s$ converges uniformly in $\orl B(z_0,r)$;
        \item[(i$'$)]In $B(z_0,R)$, $s$ is holomorphic, whose derivative is given by termwise differentiation
            \[s'(z)=\sum_{k=1}^\infty ka_k(z-z_0)^{k-1}\]
            and has the same radius of convergence;
            \ntg{The proof is to directly compare $\sum ka_k(z-z_0)^k$ with $\frac{s(z+h)-s(z)}{h}$.}
        \item[(ii)]$s$ diverges outside $\orl B(z_0,R)$;
        \item[(iii)]Nothing is said on the boundary of $B(z_0,R)$.
    }
    A function $f$ in an open set $\Omega$ is \tf{representable by power series} in $\Omega$ if in each $B(z,r)\subset\Omega$, there is a power series converging to $f$. If $f$ is representable by power series in $\Omega$, then $f\in H(\Omega)$, and its derivative $f'$ is obtained by termwise differentiation. Note $f'$ satisfies the same hypothesis, hence $f$ is differentiable infinitely many times.
\end{description}
\end{subsection}

\begin{subsection}{Propositions}
\begin{description}
\item[Structure of $H(\Omega)$]\index{$H(\Omega)$}$H(\Omega)$ is a ring (multiplication being multiplication of functions). The differential operation follows the usual rules.
\item[Power series\index{series!power}\index{power series}]See the definition section.
\end{description}
\end{subsection}
\end{section}

\begin{section}{M\"obius Transformations}
\begin{subsection}{Definitions and Properties}
\begin{description}
\item[Circles\footnote{Infinitely long straight lines are also called circles (the $\infty$) is added.} in $\orl\CC$\index{circle}]Any \tf{circle} in $\orl\CC$ can be written in the form
    \[Az\orl z+\orl Bz+B\orl z+C=0\quad(A,C\in\RR,\ B\in\CC,\ \abs{B}^2-AC>0\]
    Let $\Gamma$ be a circle and $(z_1,z_2,z_3)$ distinct points on $\Gamma$. We say this ordered triplet determines an \tf{orientation}\index{orientation} of $\Gamma$. The \tf{left side}\index{left side!of a circle} of $\Gamma$ with respect to the orientation $(z_1,z_2,z_3)$ is the set of points
    \[L=\{x\in\orl\CC:\Im(z,z_1,z_2,z_3)<0\}\]
    Respectively, if the sign is changed to $>$, then it is called the \tf{right side}\index{right side!of a circle}.
    \ntg{As an easy way to remember this, one can consider the identity map $(z,1,0,\infty)=z$. With respect to the orientation $(1,0,\infty)$, the lower half plane
        \[\Im(z,1,0,\infty)=\Im z<0\]
        is clearly the left side of $\orl\RR$.}
\item[M\"obius transformation]\index{transformation!M\"obius}\index{transformation!linear fractional}A mapping of the form
    \[f(z)=\frac{az+b}{cz+d}\quad(ad-bc\neq0)\]
    is called a \tf{M\"obius transformation} or a \tf{linear fractional transformation}. For computational convenience, it is better to write it in the matrix form
    \[f\sim\matp{a&b\\c&d\\}=A_f\quad(\det A_f\neq0)\]
    \ntg{The representation by a matrix is unique up to a multiple of a nonzero number.}
    The composite of $f\sim A_f,g\sim A_g$ is $f\comp g\sim A_fA_g$, and the inverse of $f\sim A_f$ is
    \[f^{-1}\sim A_f^{-1}=\matp{d&-b\\-c&a}\]
    \ntg{The trick to compute the inverse is: switch the diagonal and minus the sides.}
    The following types are elementary among all M\"obius transformations.
    \itm{
        \item[(i)](\tf{Parallel translation}\index{translation!parallel}) $f(z)=z+b\sim\matp{1&b\\0&1}$.
        \item[(ii)](\tf{Rotation}\index{Rotation}) $f(z)=kz\sim\matp{k&0\\0&1}$ with $\abs{k}=1$.
        \item[(iii)](\tf{Homothetic transformation}\index{transformation!homothetic}) $f(z)=kz\sim\matp{k&0\\0&1}$ with $k>0$.
        \item[(iv)](\tf{Inversion}\index{inversion}) $f(z)=1/z\sim\matp{0&1\\1&0}$.
    }
    And one can prove that any M\"obius transformation is a composite some these elementary transformations. A very special kind of M\"obius transformation is the \tf{cross ratio}\index{cross ratio}. The \tf{cross ratio} of four points $z_1,z_2,z_3,z_4$ is the image of $z_1$ under the unique M\"obius transformation which takes $z_2$ to $1$, $z_3$ to $0$ and $z_4$ to $\infty$, namely,
    \[(z,z_2,z_3,z_4)=\prn{\frac{z-z_3}{z-z_4}}\Big/\prn{\frac{z_2-z_3}{z_2-z_4}}\]
    The M\"obius transformations have the following basic properties.
    \itm{
        \item[(i)](\tf{Fixed points}) A nonidentity M\"obius transformation admits at most $2$ fixed points, namely, for $S(z)=\frac{az+b}{cz+d}$, the fixed points satisfy
            \mtha{&&f(z)=z\\
                &\iff&az+b=z(cz+d)\\
                &\iff&cz^2+(d-a)z-b=0}
            One can also include infinity in this proposition. If $\infty$ is a fixed point, then it is necessary $c=0$ and the equation becomes
            \[(d-a)z-b=0\]
            meaning $S$ admits only one more fixed point beside $\infty$.
        \item[(ii)](\tf{Uniquely determined by three points}) For any distinct points $z_2,z_3,z_4,\omega_2,\omega_3,\omega_4\in\orl\CC$, there exists a unique M\"obius transformation such that $z_2\mapsto\omega_2,\ z_3\mapsto\omega_3,\ z_4\mapsto\omega_4$.
            \prf{Let $S(z)=(z,z_2,z_3,z_4),\ M(\omega)=(\omega,\omega_2,\omega_3,\omega_4)$. Then $M^{-1}S$ satisfies the requirement. For uniqueness, if $R,S$ are M\"obius transformations that agree on three points, then $R^{-1}{S}$ is a M\"obius transformation with $3$ fixed points, hence the identity.}
        \item[(iii)](\tf{Substitution by a M\"obius transformation}) The cross ratio is invariant under a change of variable by a M\"obius transformation, i.e., if $T$ is any M\"obius transformation and $z_2,z_3,z_4$ are distinct points, then for all $z$,
            \[(z,z_2,z_3,z_4)=(Tz,Tz_2,Tz_3,Tz_4)\]
            \prf{Let $S(z)=(z,z_2,z_3,z_4)$. Consider $ST^{-1}$. It satisfies $ST^{-1}(Tz_2)=1,\ ST^{-1}(Tz_3)=0,\ ST^{-1}(Tz_4)=\infty$, hence it is the cross ratio
            \[ST^{-1}(z)=(z,Tz_2,Tz_3,Tz_4)\]
            Replacing $z$ by $Tz$ yields
            \[S(z)=(z,z_2,z_3,z_4)=(Tz,Tz_2,Tz_3,Tz_4)\]}
    }
    The following describes more about the geometric properties of M\"obius transformations.
    \itm{
        \item[(i)](\tf{Cross ratio: circle to real line}) Let $z_1,z_2,z_3,z_4$ be four distinct points in $\orl\CC$. Then
            \[(z_1,z_2,z_3,z_4)\in\RR\iff z_1,z_2,z_3,z_4\ttt{ lie on the same circle}\]
            \prf{It is equivalent to say for the M\"obius transformation $S(z)=(z,z_2,z_3,z_4)$ that
            \[S(z)\in\RR\iff z\ttt{ lies on the same circle as }z_2,z_3,z_4\]
            It is then equivalent to say for $S^{-1}$ that
            \[\omega\in\RR\iff S^{-1}(\omega)\ttt{ lies on the same cirlce as }z_2,z_3,z_4\]
            If we can show that a M\"obius transformation maps $\RR$ to a circle then we are finished.
            }
        \item[(ii)](\tf{Preserving a circle}) M\"obius transformations carries circles into circles.
        \item[(iii)](\tf{Preserving symmetric points about a circle}) If a M\"obius transformation takes the circle $\Gamma_1$ into $\Gamma_2$, then it takes a pair of symmetric points about $\Gamma_1$ into a pair about $\Gamma_2$.
        \item[(iv)](\tf{Orientation principle\index{principle!orientation}}) Let $\Gamma_1,\Gamma_2$ be two circles in $\orl\CC$ and let $T$ be a M\"obius transformation such that $T(\Gamma_1)=\Gamma_2$. Let $(z_1,z_2,z_3)$ be an orientation of $\Gamma_1$. Then the left side of $\Gamma_1$ with respect to $(z_1,z_2,z_3)$ is mapped to the left side of $\Gamma_2$ with respect to $(Tz_1,Tz_2,Tz_3)$.
    }
\end{description}
\end{subsection}

\begin{subsection}{Finding the Right M\"obius Transformations}
This section is devoted to constructing useful M\"obius transformations to provide a suitable change of variable.
\emp{(\tf{Half plane to unit disk}) The M\"obius transformation
    \[T(z)=e^{i\theta}\frac{z-a}{z+\orl a}\quad(\theta\in\RR,\ \Im a>0)\]
    carries the upper half plane to the inside of the unit disk. Conversely, all M\"obius transformations that carry the upper half plane to the inside of the unit disk have this form.
    \prf{Verifying that $T$ has the stated property is easy (it suffices to prove $z\in\orl\RR\implies\abs{T(z)}=1$). But to find all such transformations, in principle, one can find three points in $\orl\RR$ and then another three points on the unit circle, and then use the cross ratio. But the computation is too complex for our purpose and there may be redundant results. An easier way is to proceed as follows. Since a M\"obius transformation is bijective, some point must be mapped to $0$. We suppose $a:\ \Im a>0$ is such that $T(a)=0$. Then we make use of the fact that $T$ preserves symmetric points, i.e., $T(\orl a)=\infty$. This means $T$ has the form
    \[T(z)=K\frac{z-a}{z-\orl a}\]
    Finally, in order for $T$ to map $\orl\RR$ to the unit circle, we need
    \[z\in\orl\RR\implies\abs{T(z)}=\abs{K}\frac{\abs{z-a}}{\abs{z-\orl a}}=1\]
    Note that $z\in\orl\RR$ implies $\abs{z-a}=\abs{z-\orl a}$, hence it suffices that $\abs{K}=1\implies K=e^{i\theta}$.}
    \ntg{The assumption that $T(a)=0$ provides no essential information, but it does us a head start by allowing us to assume $T(z)=K\frac{z-a}{z-\orl a}$. If we had started with $T(z)=K\frac{az+b}{cz+d}$, the computation could have been very nasty.}
    It is also useful to consider its inverse
    \[T^{-1}(z)=\frac{\orl az+ae^{i\theta}}{-z+e^{i\theta}}\]
    \qed}
\emp{(\tf{Unit disk to unit disk}) The M\"obius transformation
    \[T(z)=e^{i\theta}\frac{z-a}{1-\orl az}\quad(\theta\in\RR,\ \abs{a}<1)\]

\item[Bounded variation\index{variation!bounded}]A function $\gamma:[a,b]\to\CC$ is of \tf{bounded variation} if its variation with respect to a partition $\pi$ of the parameter interval $[a,b]$ is uniformly bounded with respect to $\pi$:

\begin{subsection}{Riemann Mapping Theorem}
Let $\mcl F$ be a family of functions $D\to\CC$. We say $\mcl F$ is \tf{equicontinuous}\footnote{I call this uniformly uniformly continuous, first uniform for points and second uniform for functions.} if for any $\vare>0$, there exists $\delta>0$ such that $\abs{f(z_1)-f(z_2)}<\vare$ whenever $\abs{z_1-z_2}<\delta$ and $f\in\mcl F$. The family $\mcl F$ is called \tf{uniformly bounded} if there exists $M>0$ such that all $f\in\mcl F$ are bounded by $M$.

Before we start we need the following preparations in topology.
\lem{For any compact metric space $X$, there exists a countable Vitali covering and hence a countable dense subset of $X$.}
\prf{Consider the families of balls.
    \[\mcl{\wt{B}}_n=\{B(x,1/n):x\in X\}\]
    Each $\mcl{\wt B}_n$ is an open covering of $X$. Since $X$ is compact, we can extract a finite covering $\mcl B_n$ of $X$. Then $\mcl B=\bigcup_{n=1}^\infty \mcl B_n$ is a Vitali covering. Indeed, for any $x\in X$ and $\eta>0$, we choose $n>\frac{1}{\eta}$ then there exists a ball $B$ in $\mcl B_n$ (hence in $\mcl B$ as well) that covers $x$ with radius less than $\eta$. The centers of these balls obviously form a countable dense subset of $X$.}

\lem{Any open subset $D$ of $\RR^n$ has a compact exhaustion.}
\prf{Let $K_n=\orl{B(0,n)}\cap\{z\in D:d(z,\boun D)\geqs\frac{1}{n}\}$. Then $K_n$ is compact and $\bigcup_{n=1}^\infty K_n=D$. Moreover we have $K_n\subset\itr{K_{n+1}}$.}

\thm{\tf{(Arzel\`a-Ascoli)}For any compact metric space $X$, if $\{f_n\}$ is a sequence of continuous functions from $X$ to $\CC$ that is uniformly bounded and equicontinuous, then $\{f_n\}$ has a uniformly convergent subsequence.}
\prf{First there exists a dense subset $P$ of $X$. Next we use the diagonal method to prove that there exists a subsequence $\{f_{n_k}\}$ that converges in $P$. To prove the uniform convergence of this sequence on $X$, we use estimate
\[\abs{f_{n_k}(z)-f_{n_j}(z)}\leqs\abs{f_{n_k}(z)-f_{n_k}(a)}+\abs{f_{n_k}(a)-f_{n_j}(a)}+\abs{f_{n_j}(a)-f_{n_j}(z)}\]
We want the first and the third term to be estimated by the equicontinuity and the second term estimated by the convergence at $a$. To achieve this, we must have $\abs{a-z}<\delta$ and $a\in P$. Both can be achieved because $P$ is dense (the uniformity needs extra but little work).}

\thm{\tf{(Riemann)} Let $\Omega\subset\CC$ be a domain such that
    \itm{
        \item[(a)]$\Omega\subsetneqq\CC$
        \item[(b)]For any analytic $f:\Omega$ with $f\neq0$ everywhere in $\Omega$ there exists a branch of $\sqrt{f}$.
    }
    Then $\Omega$ is holomorphically equivalent to $D=B(0,1)$.}
\prf{Suppose $0\in\Omega$ by translation. Take $0\neq z_1\in\Omega$. Consider the family \tf{\red{(Q: How to show $\mcl A$ is nonempty)}}
    \[\mcl A=\{h:\Omega\to D,\ h(0)=0,\ h\ttt{ is analytic and univalent}\}\]
    Since each $h\in\mcl A$ is univalent and $h(0)=0$, we have $h(z_1)\neq0\implies0<\alpha=\sup_{h\in\mcl A}\abs{h(z_1)}\leqs1$. By the definition of $\alpha$, there exists a sequence $f_n\in\mcl A$ that is uniformly bounded (by assumption $\abs{h(z)}\leqs1$ and analytic). By Montel's theorem there exists subsequence $f_{n_k}$ converging to some $F(z)$ uniformly on every compact subset of $\Omega$, with $F(z)$ analytic, $F(0)=0$ and $\abs{F(z_1)}=\alpha$. Hence $F$ is not constant, then by Hurwitz's theorem \red{(Why?)} $F$ is univalent on $\Omega$.}

\cor{\tf{(Riemann)} Any simply connected domain $\Omega\subsetneqq\CC$ is holomorphically equivalent.}
\prf{One can do the square root on any simply connected domain.}

\cor{\tf{(Riemann)} Let $\Omega\subsetneqq\CC$ be a simply connect domain. Among all the holomorphic equivalences $f:\Omega\to D=B(0,1)$, the one such that $f(a)=0,\ f'(a)>0$ exists and is unique. Moreover if $F(z)=\frac{f(z)}{f'(a)}$, then $F:\Omega\to B(0,\frac{1}{f'(a)})$ is a holomorphic equivalence with $F(a)=0,\ F'(a)=1$.}
\prf{Existence: easy. Uniqueness: suppose $f,g$ both satisfies that, then consider $f\comp g^{-1}$ and $g\comp f^{-1}$.}
The radius $R=\frac{1}{f'(a)}$ of the image of $F$ above is called the \tf{radius} of $\Omega$ with respect to $a$.
