\begin{section}{Examples in Analysis}
\begin{subsection}{Common Sense---I Guess}
\begin{enumerate}
\item $x\neq k\pi,\,k\in\ns{Z}\implies\lim\limits_{n\to\infty}\sin nx\neq0$.
\end{enumerate}
\end{subsection}

\begin{subsection}{Common Series Used to Compare With}
\begin{enumerate}
\item $\sumn\frac{1}{n^\alpha}$ converges for $\alpha>1$ and diverges for $\alpha\leqs1$.
    \sprf{Integral test.}
\item $\sum\limits_{n=2}^\infty\frac{1}{n(\ln n)^\alpha}$ converges for $\alpha>1$ and diverges for $\alpha\leqs1$.
    \sprf{Integral test. Notice that
        \mth{\int\frac{1}{x(\ln x)^\alpha}\dx=\int\frac{\ud(\ln x)}{(\ln x)^\alpha}=
            \left\{\arr{{lll}\frac{(\ln x)^{1-\alpha}}{1-\alpha}+C&,&\alpha\neq1\\\ln\ln x+C&,&\alpha=1}\right.}}
    \ntg{Ermakov's test is an easier way to avoid the integral.}
\item The convergence of $\sum\limits_{n=3}^\infty\frac{1}{n(\ln n)^p(\ln\ln n)^q}$ is as follows
    \itm{
        \item $p>1\implies$ it converges for all $q$.
        \item $p=1\implies$ it converges for $q>1$.
    }
\end{enumerate}
\end{subsection}
\begin{subsection}{So Classic That They Have Names}
\begin{description}
\item[Hypergeometric series]The \udl{hypergeometric series} is defined as
    \mth{F(\alpha,\beta,\gamma,x)=1+\sum\limits_{n=1}^\infty\frac{\alpha(\alpha+1)\cdots(\alpha+n-1)\beta(\beta+1)\cdots(\beta+n-1)}{\gamma(\gamma+1)\cdots(\gamma+n-1)}\frac{x^n}{n!}}
    It is well-defined for $\abs{x}<1$ and satisfies the \udl{hypergeometric equation}
    \mth{x(x-1)y''-[\gamma-(\alpha+\beta-1)x]y'+\alpha\beta y=0}
\end{description}
\end{subsection}
\begin{subsection}{What Convergence Test Should I Apply}
\begin{description}
\item[Cauchy's test]\quad
\enu{
    \item $\sum\limits_{n=2}^\infty\frac{1}{(\ln n)^n}$
    \item $\sum\limits_{n=1}^\infty\Brk{\frac{x}{n}}^n\quad(x>0)$
    \item $\sum\limits_{n=1}^\infty\Brk{\frac{x}{a_n}}^n\quad(x>0,\,a_n>0,\,\lim\limits_{n\to\infty}a_n=a)$
    \item $\sum\limits_{n=1}^\infty\tau(n)x^n\quad(\tau(n)\ttt{ is the number of factors of }n)$
    \item $1+a+ab+a^2b+a^2b^2+\cdots+a^nb^{n-1}+a^nb^n+\cdots\quad(a,b>0,\,a\neq b)$
    }
\item[d'Alembert's test]\quad
\enu{
    \item $1+\sum\limits_{n=1}^\infty\frac{x^n}{n!}\quad(x>0)$
    \item $\sum\limits_{n=1}^\infty nx^{n-1}\quad(x>0)$
    \item $\sum\limits_{n=1}^\infty\frac{x^n}{n^s}\quad(x>0,s>0)$
    \item $\sum\limits_{n=1}^\infty n!\Brk{\frac{x}{n}}^n\quad(x>0)$
    \ntg{You have to use the nonlimit form of d'Alembert's test on this one}
    \item $1+a+ab+a^2b+a^2b^2+\cdots+a^nb^{n-1}+a^nb^n+\cdots\quad(a,b>0,\,a\neq b)$
    \ntg{Not a good choice}
    }
\item[Raabe's test]\quad
\enu{
    \item $1+\sum\limits_{n=1}^\infty\frac{(2n-1)!!}{(2n)!!}\frac{1}{2n+1}$
    \item $\sum\limits_{n=1}^\infty\frac{n!}{(x+1)\cdots(x+n)}\quad(x>0)$
    \item $\sum\limits_{n=1}^\infty\frac{n!x^n}{(x+a_1)(2x+a_2)\cdots(nx+a_n)}\quad(x>0,\,a_n>0,\,\lim\limits_{n\to\infty}a_n=a)$
    \item $\sum\limits_{n=1}^\infty\frac{1}{n!}\Brk{\frac{n}{e}}^n$
    \ntg{The limit of Raabe sequence of this series should be obtained by applying L'Hospital's rule to a corresponding function}
    }
\item[Gauss's test]\quad
\enu{
    \item The hypergeometric series.
    \item $1+\Brk{\frac{1}{2}}^p+\Brk{\frac{1\cdot3}{2\cdot4}}^p+\cdots+\Brk{\frac{(2n-1)!!}{(2n)!!}}^p+\cdots\quad(p>0)$
    }
\item[Cauchy's integral test]\quad
\enu{
    \item $\sum\limits_{n=2}^\infty\frac{1}{n\cdot\ln^{1+\sigma}n}\quad(\sigma>0)$
    \item $\sum\limits_{n=3}^\infty\frac{1}{n\cdot\ln n\cdot\ln\ln n}$
    \item $\sum\limits_{n=3}^\infty\frac{1}{n\cdot\ln n\cdot(\ln\ln n)^{1+\sigma}}\quad(\sigma>0)$
    }
\end{description}
\end{subsection}
\end{section}

\begin{section}{Counterexamples in Analysis}
\begin{subsection}{Nondecreasing alternating series that do not converge}
The following alternating series do not converge
\enu{
    \item $\frac{1}{\sqrt{2}-1}-\frac{1}{\sqrt{2}+1}+\frac{1}{\sqrt{3}-1}-\frac{1}{\sqrt{3}+1}+\cdots+\frac{1}{\sqrt{n}-1}-\frac{1}{\sqrt{n}+1}+\cdots$
    \item $\sum\limits_{n=1}^\infty\left[\frac{(-1)^{n-1}}{\sqrt{n}}+\frac{1}{n}\right]$
    \item $\sumn(-1)^n\cdot\frac{2+(-1)^n}{n}$
    }
\end{subsection}
\begin{subsection}{$\sum a_n^2$ converges but $\sum a_n$ does not}
    $\sum\limits_{n=1}^\infty\frac{1}{n^2}$ converges but $\sum\limits_{n=1}^\infty\frac{1}{n}$ does not.
\end{subsection}
\begin{subsection}{Case where comparison in the limit form fails}
    The series
    \mth{\sumn a_n=\sumn\frac{(-1)^{n-1}}{\sqrt{n}}}
    \mth{\sumn b_n=\sumn\Brk{\frac{(-1)^{n-1}}{\sqrt{n}}+\frac{1}{n}}}
    are such that
    \mth{\limn\frac{b_n}{a_n}=1}
    but the latter does not converge.
\end{subsection}
\end{section}
